using System;

class compareFiles
{
    static void Main()
    {
        int n = 0;

        StreamWriter secondInput = new StreamWriter("../../input2.txt");
        int same = 0;
        using (input)
        {
            using (secondInput)
            {
                for (string line,secondLine; (line = input.ReadLine()) != null && (secondLine = secondInput.ReadLine()) != null; n++)
                {
                    if (line == secondLine)
                    {
                        same++;
                        Console.WriteLine("Same line are: {0}", same);        

            }
        }
    }
}
